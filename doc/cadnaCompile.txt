    libcadnaOpenmpCdebug_a-cadna_intr.o \
    libcadnaOpenmpCdebug_a-cadna_math.o \
    libcadnaOpenmpCdebug_a-cadna_type.o

sudo cp libcadnaOpenmpCdebug.so /usr/local/lib/

sudo ldconfig

cd -
